Upload Files using Express and Multer

1. Create a new project folder
2. open a new terminal in project folder
3. initilize the project by running npm init -y
2. Run npm i express multer to install the required dependencies
3. Create a new javascript file called server.js
const express = require("express")
const multer = require("multer")
const upload = require({key:value})
const express = require('express')
const multer = require('multer')
const upload = multer({dest: 'uploads/'})
const app = express()
app.post('/upload', upload.any(),(req, res, next)=> {
// req.file will hold a single file
// req.body will hold the text fields, if there were any
})
